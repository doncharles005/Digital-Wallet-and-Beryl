def create_sequences(data, seq_length):
    sequences = []

        label = data[i+seq_length]  # Label corresponding to the next point

    return sequences

class UnivariateGatedTransformer(nn.Module):
    def __init__(self, input_dim, d_model, nhead, num_layers, dim_feedforward, dropout=0.1):
        super(UnivariateGatedTransformer, self).__init__()
        
        self.pos_encoder = PositionalEncoding(d_model, dropout)
        self.encoder_layer = nn.TransformerEncoderLayer(d_model=d_model, nhead=nhead, dim_feedforward=dim_feedforward)
        self.transformer_encoder = nn.TransformerEncoder(self.encoder_layer, num_layers=num_layers)
        self.fc_out = nn.Linear(d_model, input_dim)  # Predict a continuous value for regression
        
        # Gating mechanism
        self.gate_fc = nn.Linear(d_model, 1)
    
    def forward(self, src):
        # src shape: (seq_length, batch_size, input_dim)
        src = self.pos_encoder(src)
        output = self.transformer_encoder(src)
        
        # Apply gating mechanism
        gate_weights = torch.sigmoid(self.gate_fc(output.mean(dim=0)))  # Compute gating weights
        gated_output = output.mean(dim=0) * gate_weights
        
        # Final output (no softmax, as it's regression)
        output = self.fc_out(gated_output)
        return output

class PositionalEncoding(nn.Module):
    def __init__(self, d_model, dropout=0.1, max_len=5000):
        super(PositionalEncoding, self).__init__()
        self.dropout = nn.Dropout(p=dropout)

        pe = torch.zeros(max_len, d_model)
        position = torch.arange(0, max_len, dtype=torch.float).unsqueeze(1)
        div_term = torch.exp(torch.arange(0, d_model, 2).float() * (-np.log(10000.0) / d_model))
        pe[:, 0::2] = torch.sin(position * div_term)
        pe[:, 1::2] = torch.cos(position * div_term)
        pe = pe.unsqueeze(0).transpose(0, 1)
        self.register_buffer('pe', pe)

    def forward(self, x):
        x = x + self.pe[:x.size(0), :]
        return self.dropout(x)

def plot_mse_over_epochs(train_mse, test_mse, val_mse, epochs, best_epoch_train, best_epoch_test, best_epoch_val):
    plt.figure(figsize=(12, 8))
    
    # Plot MSE for training, testing, and validation
    plt.plot(epochs, train_mse, 'r-', label='Training MSE')
    plt.plot(epochs, test_mse, 'b-', label='Testing MSE')
    plt.plot(epochs, val_mse, 'g-', label='Validation MSE')
    
    # Add dotted lines to show where MSE stops declining
    plt.axvline(x=best_epoch_train, color='r', linestyle='--', linewidth=1)
    plt.axvline(x=best_epoch_test, color='b', linestyle='--', linewidth=1)
    plt.axvline(x=best_epoch_val, color='g', linestyle='--', linewidth=1)
    
    # Annotate the best performance
    plt.text(best_epoch_train, train_mse[best_epoch_train], f'Best Train: {train_mse[best_epoch_train]:.4f} @ Epoch {best_epoch_train}', color='r', verticalalignment='bottom', horizontalalignment='right')

    # Labels and legend
    plt.xlabel('Epochs')
    plt.ylabel('Mean Squared Error')
    plt.title('MSE Over Epochs for Training, Testing, and Validation')
    plt.legend()
    plt.grid(True)

class UGTN(nn.Module):
    def __init__(self, input_dim, hidden_dim, num_layers, output_dim, seq_length):
        super(UGTN, self).__init__()
        self.input_dim = input_dim
        self.hidden_dim = hidden_dim
        self.num_layers = num_layers
        self.seq_length = seq_length
        
        self.pos_encoder = nn.Linear(input_dim, hidden_dim)
        self.transformer_encoder = nn.TransformerEncoder(
            nn.TransformerEncoderLayer(d_model=hidden_dim, nhead=8), num_layers=num_layers)
        self.gate_fc = nn.Linear(hidden_dim, 1)  # Gating layer
        self.fc_out = nn.Linear(hidden_dim, output_dim)  # Output layer

    def forward(self, src):
        # src shape: (seq_length, batch_size, input_dim)
        src = self.pos_encoder(src)  # Positional encoding as a simple linear layer
        output = self.transformer_encoder(src)
        
        # Apply gating mechanism
        gate_weights = torch.sigmoid(self.gate_fc(output.mean(dim=0)))  # Compute gating weights
        gated_output = output.mean(dim=0) * gate_weights  # Element-wise multiplication with weights
        
        # Pass through final output layer
        gated_output = self.fc_out(gated_output)  # Output layer to match the target shape
        
        return output, gated_output

def entropy(attention_weights):
    """Calculates entropy of attention weights to assess focus/spread."""
    p = torch.softmax(attention_weights, dim=-1)  # Softmax to get probabilities
    log_p = torch.log(p + 1e-10)  # Log probabilities
    entropy_values = -(p * log_p).sum(dim=-1)  # Entropy calculation
    return entropy_values.mean()
